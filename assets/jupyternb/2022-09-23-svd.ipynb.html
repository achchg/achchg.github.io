def svd_decomposed_img(k, U, s, Vh):
    '''
    Recompose X with different k values: 
    k is the number of important eigenvectors in subtracting original img info
    '''
    U = U[:, :k]
    s = s[:k]
    Vh = Vh[:k, :]
    
    sigma = np.diag(s)
    X = np.dot(np.dot(U, sigma), Vh)
    
    return X
