Inverse transform sampling in code¶

A question that I read today: "How to uniformly sample from a circle?"

In [1]:
import numpy as np
import math
import matplotlib.pyplot as plt

Steps¶

  1. Derive the CDF of random variable r
  2. Use inverse transform sampling to sample r with the random uniform variale u
  3. Uniformly sample $\theta$
  4. Transform from polar to Cartesian coordinate

R = 1
N = 1000

def random_circle_sample(R, N):
    """
    Random unifromly sample N samples from a circle
    N: Number of simulation/samples
    R: radius of the circle
    """
    
    u = np.random.uniform(0, 1, N)
    theta = 2 * math.pi * np.random.uniform(0, 1, N)
    r = R * np.sqrt(u)
    
    y = r*np.sin(theta)
    x = r*np.cos(theta)
    
    plt.figure(figsize=(8, 8), dpi=80)
    plt.scatter(x, y)
    plt.show()
